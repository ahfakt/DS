7template <typename T>
8CQueue<T>::~CQueue()
9{
10 CQNode<T>* h{mHead.loadExclusively()};
11 while (h) {
12 CQNode<T>* q{h};
13 h = h->next;
14 delete q;
15 }

18template <typename T>
19bool
20CQueue<T>::enqueue(CQNode<T>* const q) noexcept
21{
22 CQNode<T>* t{mTail.load(std::memory_order::acquire)};
23
24 // Race on mTail to enqueue
25 while (!mTail.compare_exchange_weak(t, q, std::memory_order::release, std::memory_order::relaxed));
26
27 if (t) {
28 CQNode<T>* n{nullptr};
29 if (std::atomic_ref{t->next}.compare_exchange_strong(n, q, std::memory_order::release, std::memory_order::relaxed)) {
30 std::atomic_ref{mSize}.fetch_add(1, std::memory_order::relaxed);
31 return false;
32 }
33
34 t->subRef();
35 }
36
37 // Queue is empty
38 mHead.store(q, std::memory_order::release);
39 std::atomic_ref{mSize}.fetch_add(1, std::memory_order::relaxed);
40 return true;
41}

43template <typename T>
44IntrusivePtr<T>
45CQueue<T>::dequeue() noexcept
46{
47 CountedPtr<CQNode<T>> q{mHead.load(std::memory_order::acquire)};
48 CountedPtr<CQNode<T>> h{nullptr};
49
50 while (q && (h = mHead.fetchAddCount(1, std::memory_order::release))) {
51 (q = h).cnt += 1;
52 CQNode<T>* n{std::atomic_ref{h->next}.load(std::memory_order::acquire)};
53 do { // Race on mHead to dequeue
54 if (mHead.compare_exchange_weak(q, n, std::memory_order::release, std::memory_order::relaxed)) {
55 std::atomic_ref{mSize}.fetch_sub(1, std::memory_order::relaxed);
56
57 if (!n) { // Queue was empty, try to set mTail
58 CQNode<T>* t{h};
59 if (!mTail.compare_exchange_strong(t, nullptr, std::memory_order::release, std::memory_order::relaxed)) {
60 // Queue has a new item, try to set q->next before enqueue
61 if (std::atomic_ref{q->next}.compare_exchange_strong(n, q, std::memory_order::release, std::memory_order::relaxed)) {
62 // Enqueue did not set q->next yet, it may still has a reference to q
63 q->addRef(q.cnt + 1);
64 return IntrusivePtr<T>{static_cast<T*>(q->val), Offset(&CQNode<T>::val)};
65 }
66 // Enqueue has set q->next before dequeue
67 // Dequeue is responsible to update mHead
68 mHead.store(n, std::memory_order::release);
69 }
70 }
71
72 q->addRef(q.cnt);
73 return IntrusivePtr<T>{static_cast<T*>(q->val), Offset(&CQNode<T>::val)};
74 }
75 } while (q == h);
76
77 h->subRef();
78 }
79
80 return IntrusivePtr<T>{};
81}
