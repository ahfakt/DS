8template <typename T>
9struct LNode {
10 LNode* prev;
11 LNode* next;
12 Holder<T> val;
13
14 static void*
15 operator new(std::size_t, std::size_t const valSize)
16 { return ::operator new(Offset(&LNode::val) + valSize); }
17
18 // Custom delete operator to keep C++14 from using the global operator delete(void*, size_t),
19 // which would result in the compiler error:
20 // "exception cleanup for this placement new selects non-placement operator delete"
21 static void
22 operator delete(void* ptr)
23 { ::operator delete(ptr); }
24
25 explicit LNode(auto&& ... args)
26 : val(std::forward<decltype(args)>(args) ...)
27 {}
28
29 ~LNode()
30 { val->~T(); }
31};//struct LNode<T>

33template <typename T>
34void
35swap(LNode<T>* a, LNode<T>* b, LNode<T>*& head, LNode<T>*& tail) noexcept
36{
37 (a->prev ? a->prev->next : head) = b;
38 (b->next ? b->next->prev : tail) = a;
39
40 if (a->next == b) {
41 a->next = b->next;
42 b->prev = a->prev;
43 a->prev = b;
44 b->next = a;
45 } else {
46 a->next->prev = b;
47 b->prev->next = a;
48 std::swap(a->prev, b->prev);
49 std::swap(a->next, b->next);
50 }
51}

53template <typename T>
54LNode<T>*
55moveFront(LNode<T>* l, LNode<T>*& head, LNode<T>*& tail) noexcept
56{
57 if (l->prev) {
58 (l->next ? l->next->prev : tail) = l->prev;
59 l->prev->next = l->next;
60 l->next = head;
61 l->prev = nullptr;
62 head = head->prev = l;
63 }
64 return l;
65}

67template <typename T>
68LNode<T>*
69moveBack(LNode<T>* l, LNode<T>*& head, LNode<T>*& tail) noexcept
70{
71 if (l->next) {
72 (l->prev ? l->prev->next : head) = l->next;
73 l->next->prev = l->prev;
74 l->prev = tail;
75 l->next = nullptr;
76 tail = tail->next = l;
77 }
78 return l;
79}

81template <typename T>
82LNode<T>*
83moveBefore(LNode<T>* p, LNode<T>* l, LNode<T>*& head, LNode<T>*& tail) noexcept
84{
85 if (p == head)
86 return moveFront(l, head, tail);
87 if (p) {
88 (l->prev ? l->prev->next : head) = l->next;
89 (l->next ? l->next->prev : tail) = l->prev;
90 return p->prev = (l->prev = (l->next = p)->prev)->next = l;
91 }
92 return moveBack(l, head, tail);
93}

95template <typename T>
96LNode<T>*
97moveAfter(LNode<T>* p, LNode<T>* l, LNode<T>*& head, LNode<T>*& tail) noexcept
98{
99 if (p == tail)
100 return moveBack(l, head, tail);
101 if (p) {
102 (l->prev ? l->prev->next : head) = l->next;
103 (l->next ? l->next->prev : tail) = l->prev;
104 return p->next = (l->next = (l->prev = p)->next)->prev = l;
105 }
106 return moveFront(l, head, tail);
107}

109template <typename T>
110void
111qsort(LNode<T>* l, LNode<T>* r, LNode<T>*& head, LNode<T>*& tail) noexcept
112{
113 LNode<T>* p = r;
114 LNode<T>* a = l;
115 LNode<T>* b = r;
116
117 while (a != b) {
118 if (p == b) {
119 if (static_cast<T const&>(p->val) < static_cast<T const&>(a->val)) {
120 if (b == r)
121 r = a;
122 if (a == l)
123 l = b;
124 swap(a, p, head, tail);
125 b = a->prev;
126 a = p;
127 } else
128 a = a->next;
129 } else {
130 if (static_cast<T const&>(b->val) < static_cast<T const&>(p->val)) {
131 if (b == r)
132 r = a;
133 if (a == l)
134 l = b;
135 swap(p, b, head, tail);
136 a = b->next;
137 b = p;
138 } else
139 b = b->prev;
140 }
141 }
142
143 if (p != r && p->next != r)
144 qsort(p->next, r, head, tail);
145 if (l != p && l != p->prev)
146 qsort(l, p->prev, head, tail);
147}
