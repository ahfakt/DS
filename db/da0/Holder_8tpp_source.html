11template <typename T>
12struct Holder : Raw<T> {
13 Holder() noexcept = default;
14
15 explicit Holder(auto&& ... args)
16 requires Stream::InitializableFrom<T, decltype(args) ...>
17 { ::new(this) T{std::forward<decltype(args)>(args) ...}; }
18
19 explicit Holder(Stream::Source auto& input, auto&& ... args)
20 requires Stream::InitializableExtractableFrom<T, decltype(input), decltype(args) ...>
21 { input >> *::new(this) T{std::forward<decltype(args)>(args) ...}; }
22
23 explicit Holder(Stream::Source auto& input)
24 requires Stream::TriviallyExtractableFrom<T, decltype(input)>
25 { input >> *reinterpret_cast<T*>(this); }

27 template <typename ... Args, std::size_t ... I>
28 explicit Holder(Stream::Source auto& input, Pack<Args ...>& args, std::index_sequence<I ...>)
29 : Holder(input, GetAt<I>(args) ...) {}
30
31 template <typename ... Args>
32 explicit Holder(DP::CreateInfo<T, Args ...> const& createInfo, auto&& ... args)
33 { createInfo.constructor(this, std::forward<decltype(args)>(args) ...); }
